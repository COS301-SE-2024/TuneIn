export async function createAppTestingModule(): Promise<TestingModule> {
	return await Test.createTestingModule({
		controllers: [AppController],
		providers: [AppService, MyLogger],
		imports: [
			MockConfigModule,
			MockPrismaModule,
			UsersModule,
			AuthModule,
			RoomsModule,

				dest: "./uploads",
				storage: memoryStorage(),
